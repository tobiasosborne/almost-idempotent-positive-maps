\section{Wave-5--9 Lemmas Around The Dual Witness}
This section records the local machinery around a hidden row.  Let \(v\) be a hidden top
vertex, so \(p_v\) attains \(H=\max_i\distone(p_i,\conv\W)\).  Its canonical separator
\(\phi\) satisfies \(\phi\le0\) on \(\conv\W\), \(\phi(p_v)=H\), and dual norm at most one.
Set
\[
  g_i=H-\phi(p_i),\qquad \Omega_g=\max_i g_i-\min_i g_i .
\]
For a top vertex the wave-9 audit banks \(g_i\ge0\) for all rows.  Hence every \(w\in\W\)
is deep: \(g_w\ge H\).
\subsection{The exposedness LP}
Let \(F_v=\{j\ne v:\|p_j-p_v\|_1\ge\rho\}\), where \(\rho=4\tau\) and \(\kappa=\tau/4\), and
put \(d_i=p_i-p_v\).  With \(h(p_i)=u\cdot d_i\), the primal exposedness LP is
\[
  \max_{u,t}t\quad\text{s.t.}\quad
  t-u\cdot d_j\le0\ (j\in F_v),\quad -u\cdot d_k\le0,\quad u\cdot d_k\le1\ (1\le k\le n),
\]
with \(u\) free.  Its optimum is \(t^*(v)\); hidden rows have \(t^*(v)<\kappa\).
For the dual, multipliers \(\mu,\alpha,\beta\ge0\) correspond respectively to far margins,
lower boxes, and upper boxes.  The free variables \(t,u\) give
\[
  \sum_{j\in F_v}\mu_j=1,\qquad
  -\sum_{j\in F_v}\mu_jd_j-\sum_k\alpha_kd_k+\sum_k\beta_kd_k=0 .
\]
Expanding \(d_i=p_i-p_v\) and writing \(A=\sum_k\alpha_k\), \(B=\sum_k\beta_k\), gives dual
objective \(B\) and the identity below.
\begin{lemma}
\provedbadge\label{lem:diamond-dual}
The failed-exposedness dual \((\diamond)\) is exact.  If \(v\) is hidden, there are
\(\mu_j\ge0\) for \(j\in F_v\), \(\alpha_k,\beta_k\ge0\), and \(\gamma=1+A-B\), with
\[
  \sum_{j\in F_v}\mu_jp_j+\sum_k\alpha_kp_k=\sum_k\beta_kp_k+\gamma p_v,      \tag{\(\diamond\)}
\]
\[
  \sum_{j\in F_v}\mu_j=1,\qquad 1+A-B=\gamma,\qquad
  A=\sum_k\alpha_k,\quad B=\sum_k\beta_k=t^*(v).
\]
Conversely, such a witness with \(B<\kappa\) certifies failed exposedness.
\end{lemma}
\noindent Here \(\mu\) is the far-witness distribution, \(\alpha\) pays for tight lower
boxes \(h\ge0\), \(\beta\) pays for tight upper boxes \(h\le1\), and \(\gamma\) is only the
\(p_v\)-coefficient produced by expanding the shifted variables.
Pairing \((\diamond)\) with any affine \(\psi\) gives
\[
  \sum_{j\in F_v}\mu_j\psi(p_j)+\sum_k\alpha_k\psi(p_k)
  =\sum_k\beta_k\psi(p_k)+\gamma\psi(p_v).
\]
When \(t^*(v)>0\), complementary slackness puts \(\alpha\) inside the open \(\rho\)-ball of
\(v\); at \(t^*(v)=0\), far \(\alpha\)-mass is not excluded.
\begin{proposition}
\provedbadge\ The sharp exchange (W2) is
\[
  \sum_{j\in F_v}\mu_jg_j+\sum_k\alpha_kg_k=\sum_k\beta_kg_k
  \le t^*(v)\Omega_g<\kappa\Omega_g .
\]
It prices deep witness mass, but does not create it.
\end{proposition}
\begin{warning}
\provedbadge\ The witness-residual identity (W3) preserves affine residuals but loses signs.
If \(c\) is the signed index measure in \((\diamond)\), then \(P^Tc\) still has zero row
residual and zero total mass, but
its positive and negative parts need not be an exposedness dual.  This is why substituting
witness rows through \(P\) has not closed the residual.
\end{warning}
\subsection{Local consequences}
Local acronym glossary: return-flow (RF), near-delta depth (ND'), supply-forcing (SF), far-row coefficient cap (FC), carrier-blocker transpose coupling (CPL), margin cap (MC), row-witness (RW), and well-exposed-row locality (WL).
\begin{lemma}
\modauditbadge\ Return-flow from diagonal exactness (RF) holds with the audited sign caveat.
If \(P_{vv}\ge0\), \(\delta\le0.1\),
and \(\sigoff=\sum_{k\ne v}(P_{vk})_+\), then
\[
  \sum_{k\ne v}(P_{vk})_+(P_{kv})_+\ge\sigoff(1-\sigoff)-3.1\delta .
\]
When \(P_{vv}<0\), use the sign-split version; the unsplit formula is not banked.
\end{lemma}
\begin{lemma}
\modauditbadge\ Near-delta depth (ND') controls rows concentrated near their own site.  For
\(o_x=\sum_{k\ne x}|P_{xk}|\), if
\[
  P_{xx}\ge1-t,\qquad t\le t_0=\rho/2-\kappa(1+2\delta)-2\delta,\qquad
  t_0/\tau=7/4-2\tau-\delta/2,
\]
then some \(w\in\W\) satisfies \(\|p_x-p_w\|_1\le3t+5\delta\), hence
\[
  g_x\ge H-3t-5\delta .
\]
The earlier \(1.7\tau\) threshold is not banked; at \(\delta=0.05\) it is about \(1.28\tau\).
\end{lemma}
ND' explains the numerical financier identity without promoting it to exactness: a row
concentrated near a private site is forced near \(\W\), hence has deficit close to \(H\).
On the finite corner family \(H=2\delta\), the \(5\delta\) slack makes this lemma vacuous.
\begin{lemma}
\provedbadge\ Supply-forcing with the alpha-mass dichotomy (SF) converts the old
\(\alpha\)-loophole into \(A=\sum_k\alpha_k\).  With
\(S=\Av=\{k:P_{vk}>0\}\),
\[
  \sum_{j\in F_v}\mu_jp_j(S)\ge1-A\delta-t^*(v)(1+2\delta).
\]
Thus either \(A\le A_0\) and the \(\mu\)-rows are \(S\)-full up to an \(A_0\delta\) leak, or
\(A>A_0\) and \(p_v\) has an \(O(1/A_0)\)-shadow inside its own \(\rho\)-ball.  The branches
do not meet at the audited constants.
\end{lemma}
\begin{lemma}
\modauditbadge\ Far-row coefficient cap (FC) and carrier-blocker transpose coupling (CPL)
are the proved half of carrier-blocker coupling.  Every
\(\rho\)-far row \(j\) has
\[
  P_{jv}\le1-s_{\min},\qquad s_{\min}=\rho/(2+4\delta)-2\delta .
\]
If \(p_j(S)\ge1-\theta\), then
\[
  p_j(\Av)\ge s_{\min}-\theta .
\]
The CPL wording must split the \(P_{vv}=0\) and sign cases; \(P_{vv}=0\) only improves the
corner estimate.
\end{lemma}
\begin{lemma}
\provedbadge\ Margin cap from far carriers (MC) says that if \(\theta_{\mathrm{far}}\) is the
fraction of positive off-self
carrier mass of \(v\) lying in \(F_v\), then
\[
  t^*(v)\le \frac{\nu_v}{\sigoff\,\theta_{\mathrm{far}}}
  \le\frac{\delta}{\sigoff\,\theta_{\mathrm{far}}}.
\]
This caps the margin; it does not force depth.
\end{lemma}
\begin{lemma}
\provedbadge\ The generalized row-witness lemma (RW) subsumes the margin cap.  Let
\(a=P_{vv}\in[0,1)\), let
\(\sigma'=1+\nu_v-a\), and let \(\theta_{\mathrm{far}}\) be the far fraction of this positive
off-self mass.  If \(\theta_{\mathrm{far}}>0\), then
\[
  t^*(v)\le\frac{\nu_v}{\theta_{\mathrm{far}}\sigma'}.
\]
The proof is the row identity, normalized into a feasible \((\diamond)\)-dual.
\end{lemma}
\begin{lemma}
\modauditbadge\ Well-exposed-row locality (WL) has the cleaner direct-exposer proof: for
every \(w\in\W\),
\[
  \sum_{\substack{k\ne w\\ \|p_k-p_w\|_1\ge\rho}}(P_{wk})_+
  \le\frac{\nu_w}{\kappa}\le4\tau .
\]
Well-exposed vertices cannot be substantially financed by far rows.
\end{lemma}
\subsection{Wave-5--7 reductions}
Wave 5 identified the common obstruction: far rows can loiter in the top band
\(\{g<\kappa\Omega_g\}\).  The supplier-deficit route would prove Branch A from a lower
bound on carrier deficits, but the exchange controls only \(g\), not the radius or mass of
\(\alpha\).  The canonical-\(g\) energy route is dead: its own quadratic energy is bounded by
\(\delta\Omega_g^2\), too small to force an \(H^2\)-payment.
\begin{downgradedstatement}
\downgradedbadge\ The raw \(\framemat\Lambda=I\) or \(\Lambda\framemat=I\) route is not
gauge-invariant.  Its invariant content is only \(P^2=P\).  The observed binding-height
identity is LP complementary slackness, not a separate exactness law.
\end{downgradedstatement}
\begin{openproblem}
\openbadge\ Carrier-blocker coupling and aggregate pinning remain open as analytic
inequalities.  Wave 7 reduces coupling to a lower bound
\[
  \sum_b\mu_b P_{vb}\ge c\tau
\]
for far top-band blockers.  The d11 minimization did not refute this inequality, but the
ledger records it as a reduction, not as a theorem.
\end{openproblem}
\subsection{The sigma-tilde split}
The final branch variable is geometric.  Following the report convention, \(\sigtilde\) is
the positive coefficient mass of \(v\) on rows outside \(\conv\W\), including the self-row if
\(p_v\notin\conv\W\) and \(P_{vv}>0\).  The swarm meta-audit refines this with an
\(\varepsilon\)-halo: rows at \(\distone(\cdot,\conv\W)\le\varepsilon\) are automatically
\((H-\varepsilon)\)-deep.  D12 measured an index-not-in-\(W\) proxy, not this geometric test.
\begin{lemma}
\provedbadge\ The small-\(\sigtilde\) branch is closed.  For a hidden top vertex,
\[
  H\le2(1+2\delta)\max(\sigtilde,\nu_v).
\]
If \(\sigtilde\le\tau\) and \(\delta\le1/4\), then
\[
  H\le3\tau,\qquad \delta\ge H^2/9 .
\]
Equivalently, existential DMF holds on this branch with \(m_*=1\) and \(C_D=3\).
\end{lemma}
\begin{warning}
\downgradedbadge\ RW or any other feasible deep witness is not enough for existential DMF
unless it is optimal, or unless deep mass transfers to the optimal dual face.  The exchange
uses the witness's own \(B\), not automatically \(t^*(v)\).
\end{warning}
\begin{warning}
\modauditbadge\ All-shallow faces exist, but only at low height here.  A post-ledger
swarm note gives an exact rational \(5\times5\) all-shallow optimal face with hidden \(v\),
\(\sigtilde>0\), and the optimal witness supported on a far row with \(g=0\), but only at
\(H=O(\delta)\).  This is an existence certificate modulo audit; the needed exclusion is
for height of order \(\tau\), not order \(\delta\).
\end{warning}
\begin{openproblem}
\openbadge\ The remaining open is sharp: exclude hidden top vertices with
\[
  \sigtilde>\tau,\qquad H>B\tau
\]
for a universal \(B\).  This is signed quantitative Baake--Sumner stability for shallow
hidden top-band webs.  The \(\delta=0\) anchor is banked, but the quantitative theorem is
open; pushed-witness cleanup cannot finish it because negative leakage breaks dual
optimality.
\end{openproblem}
\begin{numericalresult}
\numbadge\ The empirical record is now framed as linear evidence.  The advertised flat floor
was never verified below about \(\delta=0.06\), and d13's final outcome (c) found no verified
hidden-top-vertex floor instance for \(\delta\le10^{-2}\).  Across the verified record,
heights track \(H\approx2\delta\).
\end{numericalresult}
